```
function apromise(t){
    return new Promise(function(resolve,reject){
        setTimeout(resolve,t)
    })
}
function stat(){
    apromise(3000).then(function(){
        console.log('yellow');
        return apromise(1000);
    }).then(function(){
        console.log('green');
        return apromise(2000);
    }).then(function(){
        console.log('red');
        stat();
    })
}
stat();

依次执行：
yellow
green
red
```

```
var n = 0;
function time(a){
    var timeOut = setTimeout(function(){
        n++;
        console.log(a+n);  //会每隔1s依次输出a1 a2 a3 一直到a10
        time(a);
    },1000);
    if(n>=10){
        clearTimeout(timeOut);
    }
}
time('a');
```

```
闭包：setTimeout第一个参数调用函数不能传递参数，可以通过闭包来实现。
function fa(a,b){
    return (function(){
        console.log(a+b);
    });
}
var f = fa(1,2);
setTimeout(f,2000)
```
