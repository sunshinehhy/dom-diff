## const
```
const arr = [1,1,1];
arr[1] = 3;
console.log(arr);  //[1,3,1]

const bb = 1
bb=2
console.log(bb);  //VM98:1 Uncaught TypeError: Assignment to constant variable.
```

## 闭包

```
function test(arr){

  for (var i=0;i<arr.length;i++){
    setTimeout(function(){
      console.log(i);  //输出4个4
    },0)
  }
 console.log(arr[i]);  //undefinded 因为i为5，超过了arr的长度，找不到值
}
test([1,2,3,4]);

先输出undefinded ，再输出4个4
```
```
function test(arr){
  for (var i=0;i<arr.length;i++){
    (function(j){setTimeout(function(){
      console.log(j);
    },0)}(i))
  }
 console.log(arr[i]);
}
test([1,2,3,4]);
先输出undefinded ，再输出1 2 3 4
```

## this
```
var a = {
  name:'hhy',
  getName:function(){
    return this.name;
  }
};
console.log(a.getName());  //hhy
var func = a.getName;
console.log(func());// Uncaught TypeError: Cannot read property 'name' of undefined
```
