## 检查基本数据类型和引用数据类型通用方法
```
var a = Object.prototype.toString;

console.log(a.call("aaa"));  // [object String]
console.log(a.call(1));  //[object Number]
console.log(a.call(true)); //[object Boolean]
console.log(a.call(null)); //[object Null]
console.log(a.call(undefined)); //[object Undefined]
console.log(a.call([])); //[object Array]
console.log(a.call(function() {})); //[object Function]
console.log(a.call({}));//[object Object]
```

## instanceof
```
 console.log("1" instanceof String);   //false
 console.log(1 instanceof Number);  //false
 console.log(true instanceof Boolean);  //false
 console.log(null instanceof Null);  // Uncaught ReferenceError: Null is not defined
 console.log(undefined instanceof Undefined);  //Uncaught ReferenceError: Undefined is not defined
 console.log([] instanceof Array);  //true
 console.log(function(){} instanceof Function);  //true
 console.log({} instanceof Object);  //true
```
不能判断null 和 undefined

```
    console.log(("1").constructor === String);  //true

    console.log((null).constructor === Null); //Uncaught TypeError: Cannot read property 'constructor' of null
    console.log((undefined).constructor === Undefined); //Uncaught TypeError: Cannot read property 'constructor' of undefined
    console.log(([]).constructor === Array);  //true
    console.log((function() {}).constructor === Function);  //true
    console.log(({}).constructor === Object);  //true
```
不能判断null 和 undefined

```
function Fn(){};
Fn.prototype=new Array();  //相当于重写了原型对象
var f=new Fn();

console.log(f.constructor===Fn);  //false
console.log(f.constructor===Array);  //true
console.log(Fn.prototype.constructor === Array);  //true

function Fn(){};
var f=new Fn();
console.log(f.constructor===Fn);  //true
console.log(f.constructor===Array);  //false
console.log(Fn.prototype === f);  //false
console.log(Fn.prototype.constructor === Fn);  //true
console.log(f.constructor === Fn);   //true
```
声明了一个构造函数，并且把他的原型指向了Array的原型，所以这种情况下，constructor也显得力不从心了。
